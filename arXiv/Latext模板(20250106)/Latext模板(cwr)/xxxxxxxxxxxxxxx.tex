class L_psnr(nn.Module):
   def __init__(self):
     super(L_psnr, self).__init__()
  
   def forward(self, org, enhenced):
     mse = torch.mean((org / 255. - enhenced / 255.) ** 2)
     if mse < 1.0e-10:
        return 100
     PIXEL_MAX = 1       
     return 20 * math.log10(PIXEL_MAX / math.sqrt(mse))
\end{lstlisting}
